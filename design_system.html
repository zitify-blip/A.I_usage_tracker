8. 사용 가이드 (코드)

XAML — 정적 적용

<Button Content="새로고침" Style="{StaticResource Btn}"        Click="Refresh_Click"/>
<Button Content="로그인"   Style="{StaticResource BtnPrimary}" Click="Login_Click"/>
<Button Content="로그아웃" Style="{StaticResource BtnDanger}"  Click="Logout_Click"/>
<Button Content="취소"     Style="{StaticResource BtnGhost}"   Click="Cancel_Click"/>

C# — 동적 토글

LoginBtn.Content = isLoggedIn ? "Logout" : "Login";
LoginBtn.Style = (Style)FindResource(isLoggedIn ? "BtnDanger" : "BtnPrimary");

C# — 테마 인식 색상 (UsageColor 등)

// BR(key)/CR(key) 헬퍼는 현재 테마의 ResourceDictionary에서 lookup
private static SolidColorBrush BR(string key) =>
    System.Windows.Application.Current.Resources[key] as SolidColorBrush ?? B("#888");

// 예: 신호등 색
StatusLabel.Foreground = BR(pct >= 90 ? "StatusBadBrush" :
                            pct >= 70 ? "StatusWarnBrush" :
                                        "StatusGoodBrush");

WPF ResourceDictionary 위치: Styles/DarkTheme.xaml · Styles/DogTheme.xaml — App.xaml에서 한 번만 merge.
